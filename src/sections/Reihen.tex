\section{Reihen}

\begin{itemize}
    \item Notwendiges Kriterium für die Konvergenz einer Reihe:
    \begin{align*}
        \sum_{n=0}^{\infty} a_k \quad \rightarrow \quad \lim_{k \to \infty}  a_k = 0
    \end{align*}
    \item Vergleichskriterium: 
        \begin{itemize}
            \item Majorantenkirterium:
            \begin{align*}
                |a_k| \leq b_k \quad \text{und} \quad \sum_{k}^{\infty} b_k < \infty \quad \Rightarrow \quad \sum_{k}^{\infty} a_k < \infty
            \end{align*}
            \item Minorantenkriterium:
            \begin{align*}
                |a_k| \geq b_k \quad \text{und} \quad \sum_{k}^{\infty} b_k = \infty \quad \Rightarrow \quad \sum_{k}^{\infty} a_k = \infty
            \end{align*}
        \end{itemize}
    \item Wurzelkriterium:
    \begin{align*}
        \lim_{k \to \infty} \sqrt[k]{|a_k|} = L ,\quad L < 1 \Rightarrow \sum_{k}^{\infty} a_k < \infty, \quad L > 1 \Rightarrow \sum_{k}^{\infty} a_k = \infty, \quad L = 1 \text{ unbestimmt}
    \end{align*}
    \item Quotientenkriterium:
    \begin{align*}
        \lim_{k \to \infty} \left| \frac{a_{k+1}}{a_k} \right| = L ,\quad L < 1 \Rightarrow \sum_{k}^{\infty} a_k < \infty, \quad L > 1 \Rightarrow \sum_{k}^{\infty} a_k = \infty, \quad L = 1 \text{ unbestimmt}
    \end{align*}
    \item Leibniz-Kriterium:
    \begin{align*}
        \sum_{k=0}^{\infty} (-1)^k a_k \quad \text{mit} \quad a_k > 0, \quad a_{k+1} < a_k, \quad \lim_{k \to \infty} a_k = 0 \Rightarrow \sum_{k=0}^{\infty} (-1)^k a_k \text{ konvergiert}
    \end{align*}
\end{itemize}

\textbf{Potenzreihen:}
\begin{align*}
    \sum_{k=0}^{\infty} a_k (x - x_0)^k \quad \text{mit} \quad a_k \in \mathbb{R}, \quad x_0 \in \mathbb{R} , \quad x_0 \hat{=} \text{Entwicklungspunkt}
\end{align*}

\textbf{Konvergenzradius:}
\begin{align*}
    R = \lim_{k \to \infty} \frac{1}{\sqrt[k]{|a_k|}} = \lim_{k \to \infty} \left| \frac{a_k}{a_{k+1}} \right| \quad \text{mit} \quad |x - x_0| < R \Rightarrow \text{Potenzreihe konvergiert}
\end{align*}

\textbf{Integration und Differentiation von Potenzreihen:}
\begin{align*}
    \sum_{k=0}^{\infty} a_k (x - x_0)^k \quad &\Rightarrow \quad \int_{x_0}^{x} \sum_{k=0}^{\infty} a_k (t - x_0)^k dt = \sum_{k=0}^{\infty} \frac{a_k}{k+1} (x - x_0)^{k+1} + C\\
    &\Rightarrow \quad \frac{d}{dx} \sum_{k=0}^{\infty} a_k (x - x_0)^k = \sum_{k=1}^{\infty} k a_k (x - x_0)^{k-1}
\end{align*}
Beachte, dass bei der Differentiation evtl. die untere Grenze der Summe angepasst werden muss. (kein -1 im Exponenten)

\textbf{Einsetzen in Potenzreihen:}
\begin{align*}
    &f(x) = \sum_{k=0}^{\infty} a_k x^k \quad \text{mit Konvergenzradius} R\\
    \Rightarrow \quad & f(Cx) = \sum_{k=0}^{\infty} a_k C^k x^k \quad \text{mit Konvergenzradius} \frac{R}{|C|} \\
    \Rightarrow \quad & f(x^m) = \sum_{k=0}^{\infty} a_k x^{mk} \quad \text{mit Konvergenzradius} \sqrt[m]{R} \qquad m \in \mathbb{N}\\
\end{align*}

\textbf{Addition und Subtraktion von Potenzreihen:}
\begin{align*}
                      &f(x) = \sum_{k=0}^{\infty} a_k (x - x_0)^k \quad \text{mit } R_1 \quad \text{und} \quad g(x) = \sum_{k=0}^{\infty} b_k (x - x_0)^k \quad \text{mit } R_2 \\
    \Rightarrow \quad &f(x) \pm g(x) = \sum_{k=0}^{\infty} (a_k \pm b_k) (x - x_0)^k \\
    \quad \text{mit Konvergenzradius } &R = \min(R_1, R_2) \text{, wenn } R_1 \neq R_2\\
                                      &R \geq R_1 \text{, wenn } R_1 = R_2
\end{align*}

\textbf{Multiplikation von Potenzreihen:}
\begin{align*}
    &f(x) = \sum_{k=0}^{\infty} a_k (x - x_0)^k \quad \text{mit } R_1 \quad \text{und} \quad g(x) = \sum_{k=0}^{\infty} b_k (x - x_0)^k \quad \text{mit } R_2\\
    \Rightarrow \quad &f(x) g(x) = \sum_{k=0}^{\infty} c_k (x - x_0)^k \quad \text{mit} \\
    &c_k = \sum_{j=0}^{k} a_j b_{k-j} = a_0 b_k + a_1 b_{k-1} + ... + a_k b_0\\
    \quad \text{mit Konvergenzradius } &R \geq \min(R_1, R_2)
\end{align*}

\textbf{Division von Potenzreihen:}
\begin{align*}
    &f(x) = \sum_{k=0}^{\infty} a_k (x - x_0)^k \quad \text{mit } R_1 \quad \text{und} \quad g(x) = \sum_{k=0}^{\infty} b_k (x - x_0)^k \quad \text{mit } R_2\\
    \Rightarrow \quad &\frac{f(x)}{g(x)} = \sum_{k=0}^{\infty} c_k (x - x_0)^k \quad \text{mit} \\
    &c_k = \frac{1}{b_0} \left( a_k - b_1 c_{k-1} - b_2 c_{k-2} - ... - b_{k-1} c_0 \right)\\
    \quad \text{mit Konvergenzradius } &R = \min(R_1, R_2) \text{, wenn } R_1 \neq R_2\\
                                      &R \geq R_1 \text{, wenn } R_1 = R_2
\end{align*}

\textbf{Taylorreihe:}
\begin{align*}
    T_{x_0}^n f(x) = \sum_{k=0}^{n} \frac{f^{(k)}(x_0)}{k!} (x - x_0)^k
\end{align*}

\textbf{Taylorformel:}
\begin{align*}
    f(x) =& T_{x_0}^n f(x) + R_n(f,x_0) \quad \text{mit} \quad R_n(f,x_0) = \frac{f^{(n+1)}(\xi)}{(n+1)!} (x - x_0)^{n+1} \quad \text{für} \quad \xi \in (x_0, x)\\
    \text{Außerdem gilt: }& f'(\xi) = \frac{f(x) - f(x_0)}{x - x_0} \quad \text{für} \quad \xi \in (x_0, x)
\end{align*}

\textbf{Wichtige Taylorreihen:}
\begin{align}
    e^x =& \sum_{k=0}^{\infty} \frac{x^k}{k!}\\
    \ln(x) =& \sum_{k=1}^{\infty}  \frac{(-1)^{k+1}}{k}(x-1)^k\\
    \sin(x) =& \sum_{k=0}^{\infty} (-1)^k \frac{x^{2k+1}}{(2k+1)!}\\
    \cos(x) =& \sum_{k=0}^{\infty} (-1)^k \frac{x^{2k}}{(2k)!}\\
    \frac{1}{1-qx} =& \sum_{k=0}^{\infty} q^k x^k \quad \text{für} \quad |qx| < 1\\
    {(1+x)}^{\alpha} =& \sum_{k=0}^{\infty} \binom{\alpha}{k} x^k \quad \text{für} \quad |x| < 1
\end{align}
